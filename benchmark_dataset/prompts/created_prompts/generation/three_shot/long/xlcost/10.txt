    last_index = (n // 2 + 1) if n % 2 == 1 else (n // 2)
    for i in range(last_index):
        if arr[i] == x and arr[i + n // 2] == x:

def findPair(arr, n):
    size = len(arr)
    i, j = 0, 1

    while i < size and j < size:
        if i != j and arr[j] - arr[i] == n:
            print(f"Pair found ({arr[i]}, {arr[j]})")
            return True
        elif arr[j] - arr[i] < n:
            j += 1
        else:
            i += 1

    print("No pair found")
    return False
